126     * @param {Array} commandList Messages list to send
127     * @returns {void}
128     */
129    function sendToClient (commandList)
130    {
131       var msgId = newGuid() ;
132       var msg = commandList.join("\0") ;
133       var msgLenth = msg.length ;
134       if (msgLenth > 1000)
135       {
136          var part ;
137          var partNum = 1 ;
138          var partLen ;
139          while (msgLenth > 0)
140          {
141             part = msg.substring(0, 1000) ;  // 0..999
142             msg = msg.substring(1000) ;      // 1000..end
143 
144             msgLenth -= 1000 ;
145             partLen = part.length ;
146             if (partLen >= 1000)
147                addMessage ({msgId:msgId, msg:part, partNum:partNum}) ;
148             else
149                addMessage ({msgId:msgId, msg:part, partNum:'Last'} ) ;
150             partNum++;
151          }
152       } else {
153          addMessage ({msgId:msgId, msg:msg, partNum:''}) ;
154       }
155 
156    } ;

217          // kill current script
218          headID.removeChild (ttraceScript) ;

222          runScript(objMessage);

274    /** return the function of a code fragment 
275     * @param {Object} fctName code fragment
276     * @returns {string} function name
277     */
278    function getFunctionName(fctName)
279    {
280       fctName = rTrim(lTrim(fctName)) ;
281       // [ecmascript code] or function MyClass()...
282       if (fctName.indexOf('[ecmascript code]') == 0)
283          return '?' ;
284 
285       if (fctName.indexOf('function ') == 0)// 0..8
286       {
287          var p = 9 ;
288          var endClassName = 1000 ;
289          while (fctName.charAt(p) == ' ')
290             p++;
291          // search '(' char
292          for(var c = p ; c < fctName.length; c++)
293          {
294             var ch = fctName.charAt(c) ;
295             if (ch=='(') {
296                endClassName = c;
297                break;
298             }
299          }
300          if (c == p)  // nothing before ()
301             return '?';
302 
303          return fctName.substring(9,endClassName);
304       }
305       var pos = fctName.indexOf("{") ;   // search for function body
306       if (pos >= 0)
307       {
308          if (pos ==0)
309             return "<unnamed function>" ;
310 
311          // remove function body
312          fctName = fctName.substr(0,pos) ;
313 
314          pos = fctName.indexOf("(") ;   // remove function parameters
315          if (pos != -1)
316             fctName = fctName.substr(0,pos) ;
317 
318          if (fctName.substr(0,8) == "function")
319             fctName = fctName.substr(8,fctName.length) ;
320 
321          return fctName ;
322       }
323 
324       return fctName ;
325    }

328    /** Convert a color coded as a string to a RGB integer. the color parameter can be #RRGGBB or RGB(r,g,b). 
329     * Browser colors like "yellow" or "red" are not suported (return black (0,0,0))
330     * @param {string} color Color to convert
331     * @returns {integer} RGB
332    */   
333    function rgbToBgr(color)
334    {
335       var r ;
336       var g ;
337       var b ;
338 
339       var hexString = rTrim(lTrim(color)).toUpperCase() ;
340       if (hexString.charAt(0) == '#')
341       {
342          // decompose #RRGGBB
343          //           0123456
344          r = parseInt(hexString.substring (1,3),16) ;   //  1,2 (char 3 is not included)
345          g = parseInt(hexString.substring (3,5),16) ;   //  3,4 (char 5 is not included)
346          b = parseInt(hexString.substring (5,7),16) ;   //  5,6 (char 7 is not included)
347       } else if (hexString.indexOf('RGB') == 0) {
348          // decompose RGB(r,g,b) or RGB (r,g,b)
349          //           0123          01234
350 
351          // start after the RGB word and skip spaces
352          var p = 3 ;
353          while (hexString.charAt(p) == ' ')
354             p++;
355 
356          // check the '(' and the ')'
357          if (hexString.charAt(p) != '(' &&
358              hexString.charAt(hexString.length-1) != ')')
359             return 0 ;  // error : use black
360 
361          // remove '(' and ')'
362          var block = hexString.substring(p+1,hexString.length-1) ; // r,g,b
363          var rgb = block.split(',')
364          if (rgb.length != 3)
365             return 0 ;  // error : use black (0,0,0)
366          r = parseInt(rgb[0]) ;
367          g = parseInt(rgb[1]) ;
368          b = parseInt(rgb[2]) ;
369       } else {
370          return 0 ;  // unknow color , use black (0,0,0)
371       }
372 
373       return (b << 16) + (g  << 8) + r ;
374    }

477    /** Get the class name of an object
478    * @param {Object} obj Object to get class name
479    * @returns {string} class name
480    */
481    function getClassName(obj)
482    {
483       if (obj === null)
484          return "null" ;
485 
486       var type = typeof (obj) ;
487 
488       // if not vague, return typeof(obj)
489       if (type != "object")
490          return type ;
491 
492       // if a property "classname" or "className" of type string exist, return it ;
493       var objClassname = obj.classname || obj.className;
494       if (typeof(objClassname) == "string") {
495          return objClassname ;
496       }
497 
498       try
499       {
500          if (obj.nodeName){
501             switch(obj.nodeType){
502                case  0 : return 'NODE_INVALID' ;
503                case  1 : return 'NODE_ELEMENT' ;
504                case  2 : return 'NODE_ATTRIBUTE' ;
505                case  3 : return 'NODE_TEXT' ;
506                case  4 : return 'NODE_CDATA_SECTION' ;
507                case  5 : return 'NODE_ENTITY_REFERENCE' ;
508                case  6 : return 'NODE_ENTITY' ;
509                case  7 : return 'NODE_PROCESSING_INSTRUCTION' ;
510                case  8 : return 'NODE_COMMENT' ;
511                case  9 : return 'NODE_DOCUMENT' ;
512                case 10 : return 'NODE_DOCUMENT_TYPE' ;
513                case 11 : return 'NODE_DOCUMENT_FRAGMENT' ;
514                case 12 : return 'NODE_NOTATION' ;
515             }
516          }
517          if (typeof obj.length == 'number'){
518             if (obj.item)   return 'collection';
519             if (obj.callee) return 'arguments';
520          }
521       } catch (e) {
522         return e.message;
523       }
524 
525       var protoString = Object.prototype.toString.apply(obj) ;
526       if (protoString.substr(0,7) == "[object" ||
527           protoString.substr(0,7) == "[Object")
528       {
529          protoString = protoString.substring(7,protoString.length-1);  // remove "[object" and "]"
530          protoString = protoString.replace(/^\s*|\s*$/g, "");  // // Strip leading and trailing white-space
531 
532          if (protoString == "")
533             protoString = "Object" ;
534 
535          // if not vague, return protoString
536          if (protoString.toLowerCase() != "object")
537             return protoString ;
538       }
539 
540       try   // getting "constructor" property can generate exception under firefox
541       {
542          if ("constructor" in obj)
543          {
544             // if constructor don't have prototype : unknow type
545             if (typeof(obj.constructor) == "undefined")
546                return protoString ;
547 
548             switch(obj.constructor){
549                case Array:  return 'Array';
550                case RegExp: return 'Regexp';
551                //case Class:  return 'Class';
552             }
553             // constructor sould be a function. if constructor is an object, return protoString
554             if (obj.constructor == Object)
555                return protoString ;
556 
557             // if constructor don't have prototype : unknow type
558             if (typeof(obj.constructor.prototype) == "undefined")
559                return protoString ;
560 
561             var propType = obj.constructor.toString() ;
562             if (propType.length == 0)
563                return protoString ;
564 
565             var pos = propType.indexOf("{") ;   // search for function body
566             if (pos ==-1)
567                return propType ;
568             if (pos ==0)
569                return "<unnamed constructor>" + protoString ;
570 
571             propType = propType.substr(0,pos) ;
572 
573             pos = propType.indexOf("(") ;   // remove function parameters
574             if (pos != -1)
575                propType = propType.substr(0,pos) ;
576 
577             if (propType.substr(0,8) == "function")
578                propType = propType.substr(8,propType.length) ;
579 
580             return propType ;
581 
582          }
583       } catch (e) {
584         return protoString ;
585       }
586       return protoString ;
587    }
